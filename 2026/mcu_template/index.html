cmake复制粘贴

                
cmake_minimum_required(VERSION 3.11)

# ----------------------------------------
# complier
# ----------------------------------------

set(CMAKE_SYSTEM_NAME Generic)
set(CMAKE_EXECUTABLE_SUFFIX ".elf")
set(CMAKE_SYSTEM_VERSION 1)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

set(CMAKE_C_COMPILER arm-none-eabi-gcc)
set(CMAKE_CXX_COMPILER arm-none-eabi-gcc)
set(CMAKE_ASM_COMPILER  arm-none-eabi-gcc)
set(CMAKE_AR arm-none-eabi-ar)
set(CMAKE_OBJCOPY arm-none-eabi-objcopy)
set(CMAKE_OBJDUMP arm-none-eabi-objdump)
set(SIZE arm-none-eabi-size)
set(CMAKE_TRY_COMPILE_TARGET_TYPE STATIC_LIBRARY)

project(stm32_test C ASM)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_C_STANDARD 11)

# ----------------------------------------
# arm m4
# ----------------------------------------
add_compile_definitions(ARM_MATH_CM4;ARM_MATH_MATRIX_CHECK;ARM_MATH_ROUNDING)
add_compile_options(-mfloat-abi=hard -mfpu=fpv4-sp-d16)
add_link_options(-mfloat-abi=hard -mfpu=fpv4-sp-d16)

add_compile_options(-mcpu=cortex-m4 -mthumb)
add_compile_options(
    -ffunction-sections
    -fdata-sections
    -fno-common
    -Wall
    -fno-exceptions
    -Wextra
)

# Enable assembler files preprocessing
add_compile_options($<$:-x$assembler-with-cpp>)
set(CMAKE_BUILD_TYPE "Debug")

if ("${CMAKE_BUILD_TYPE}" STREQUAL "Release")
    message(STATUS "Maximum optimization for speed")
    add_compile_options(-Ofast)
elseif ("${CMAKE_BUILD_TYPE}" STREQUAL "RelWithDebInfo")
    message(STATUS "Maximum optimization for speed, debug info included")
    add_compile_options(-Ofast -g)
elseif ("${CMAKE_BUILD_TYPE}" STREQUAL "MinSizeRel")
    message(STATUS "Maximum optimization for size")
    add_compile_options(-Os)
else ()
    message(STATUS "Minimal optimization, debug info included")
    add_compile_options(-Og -g)
endif ()

# ----------------------------------------
# code
# ----------------------------------------

file(GLOB HAL_SRC "hal/Src/*.c")
add_executable(stm32_test.elf
    ${HAL_SRC}
    src/main.c
    src/startup_stm32f429xx.s
    src/stm32f4xx_it.c
    src/syscalls.c
    src/sysmem.c
    src/system_stm32f4xx.c
)

target_include_directories(stm32_test.elf
PUBLIC
    CMSIS/Core/Include
    CMSIS/Device/ST/STM32F4xx/Include
    hal/Inc
    src
)

target_compile_definitions(stm32_test.elf
PUBLIC
    STM32F429xx=1
    USE_FULL_ASSERT=1
)

target_link_options(stm32_test.elf
PUBLIC
    -Wl,-gc-sections,--print-memory-usage,-Map=${PROJECT_BINARY_DIR}/${PROJECT_NAME}.map
    -mcpu=cortex-m4
    -mthumb
    -T "${CMAKE_CURRENT_SOURCE_DIR}/src/ld.ld"
)

# ----------------------------------------
# elf to bin
# ----------------------------------------
add_custom_command(TARGET stm32_test.elf POST_BUILD
        COMMAND ${CMAKE_OBJCOPY} -Oihex $ stm32_test.hex
        COMMAND ${CMAKE_OBJCOPY} -Obinary $ stm32_test.bin
        COMMENT "Building ${HEX_FILE}
Building ${BIN_FILE}"
)

openocd 文件

                
# 选择调试器接口: CMSIS-DAP
source [find interface/cmsis-dap.cfg]

# 选择传输协议: SWD (DAP 通常支持 JTAG 和 SWD，STM32 常用 SWD)
transport select swd

# 选择目标芯片: STM32F4 系列
source [find target/stm32f4x.cfg]

# 选填：设置下载速率 (单位 kHz)
# 如果线缆较长或信号不稳定，可以降低这个数值
adapter speed 2000

# 选填：复位配置
# srst_only 表示只有系统复位，connect_assert_srst 表示连接时拉低复位引脚
# 对于很多便宜的 DAP 调试器，如果不接 RST 引脚，可以注释掉这行
# reset_config srst_only
